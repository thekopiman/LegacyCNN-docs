76{
77 resetMask();
78
79 bool ExistNonZero = false;
80 for (int i = 0; i < channel_in; i++)
81 {
82 if (lengths[i] > 0)
83 {
84 ExistNonZero = true;
85 }
86 }
87
88 assert(ExistNonZero && "Make sure that lengths contain at least a Natural number to prevent division by 0.");
89
90 // Mask here has a dim of [N, C, L] instead of the one [N, 1, L]. Where N = 1
91 // This is because Hadamard Product in MatrixFunctions requires same dimensions.
92
93 // Length to mask
94 // https://speechbrain.readthedocs.io/en/latest/_modules/speechbrain/dataio/dataio.html#length_to_mask
95
96 for (int i = 0; i < channel_in; i++)
97 {
98 for (int j = 0; j < std::ceil(lengths[i] * input_width) && j < input_width; j++)
99 {
100 this->mask[i][j] = 1;
101 }
102 this->total[i] = MatrixFunctions::Sum(this->mask[i]);
103 }
104
105 // mask / total
106
107 for (int i = 0; i < channel_in; i++)
108 {
109 for (int j = 0; j < input_width; j++)
110 {
111 this->mask[i][j] /= this->total[i];
112 }
113 }
114
115 T s[channel_in][input_width];
116 MatrixFunctions::HadamardProduct(input, this->mask, s);
117
118 for (int i = 0; i < channel_in; i++)
119 {
120 mean[i][0] = MatrixFunctions::Sum(s[i]) / this->total[i];
121 }
122
123 layer0.forward(mean, temp);
124 ActivationFunctions::ReLU(temp);
125 // We reuse mean to save space
126 layer1.forward(temp, mean);
127 ActivationFunctions::Sigmoid(mean);
128 // Reshape mean[channel_in][1] to temp2[channel_in]
129 MatrixFunctions::Reshape(mean, temp2);
