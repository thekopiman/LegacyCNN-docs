255template <int channels, int attention_channels, int input_width, int out_width, typename T>
256void ASP<channels, attention_channels, input_width, out_width, T>::compute_statistics(T (&x)[channels][input_width], T (&m)[channels][input_width])
257{
258 T temp[channels][input_width];
259
260 // Obtain Mean here
261 MatrixFunctions::HadamardProduct(x, m, temp);
262 for (int i = 0; i < channels; i++)
263 {
264 this->mean[i] = MatrixFunctions::Sum(temp[i]);
265 }
266
267 // Obtain Population Std here
268
269 for (int i = 0; i < channels; i++)
270 {
271 for (int j = 0; j < input_width; j++)
272 {
273 x[i][j] -= this->mean[i]; // x - mean
274 x[i][j] *= x[i][j]; // pow(2)
275 }
276 }
277
278 MatrixFunctions::HadamardProduct(x, m, temp);
279 for (int i = 0; i < channels; i++)
280 {
281 this->std[i] = MatrixFunctions::Clamp(MatrixFunctions::Sum(temp[i]), (T)1e-12);
282 this->std[i] = std::sqrt(this->std[i]);
283 }
284}
