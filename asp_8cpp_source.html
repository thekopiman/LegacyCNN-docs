101template <int channels, int attention_channels, int input_width, int out_width, typename T>
102void ASP<channels, attention_channels, input_width, out_width, T>::compute_statistics(T (&x)[channels][input_width], T (&m)[channels][input_width])
103{
104 T temp[channels][input_width];
105
106 // Obtain Mean here
107 MatrixFunctions::HadamardProduct(x, m, temp);
108 for (int i = 0; i < channels; i++)
109 {
110 this->mean[i] = MatrixFunctions::Sum(temp[i]);
111 }
112
113 // Obtain Population Std here
114
115 for (int i = 0; i < channels; i++)
116 {
117 for (int j = 0; j < input_width; j++)
118 {
119 x[i][j] -= this->mean[i]; // x - mean
120 x[i][j] *= x[i][j]; // pow(2)
121 }
122 }
123
124 MatrixFunctions::HadamardProduct(x, m, temp);
125 for (int i = 0; i < channels; i++)

127 this->std[i] = MatrixFunctions::Clamp(MatrixFunctions::Sum(temp[i]), (T)1e-12);
128 this->std[i] = std::sqrt(this->std[i]);
